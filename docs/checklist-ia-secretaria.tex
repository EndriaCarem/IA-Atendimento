% =====================================================================
\section{Pr\'e-requisitos de ambiente}
\begin{itemize}
  \item WhatsApp conectado de verdade (n\~ao ``open'' fantasma) --- validar recebendo/enviando 1 msg \sparc
  \item N\'umero do bot dedicado (n\~ao aberto no celular ao mesmo tempo) --- evita queda 401/413 \sfalta
  \item IA Ativa (toggle ligado no painel) \sok
  \item Sauda\c{c}\~ao configurada no painel (aba Comportamento) \sok
  \item Endere\c{c}o, procedimentos e conv\^enios cadastrados \sok
\end{itemize}

% =====================================================================
\section{Atendimento via WhatsApp (linguagem natural)}
\begin{itemize}
  \item Responde ``oi'' com a sauda\c{c}\~ao configurada no front (nome certo da cl\'inica) \sok
  \item N\~ao repete a sauda\c{c}\~ao nas mensagens seguintes \sok
  \item Entende a inten\c{c}\~ao sem menu/bot\~ao (``quero marcar'', ``preciso remarcar'') \sok
  \item Usa o primeiro nome do paciente ap\'os identific\'a-lo \sok
  \item Tom acolhedor, respostas curtas \sok
\end{itemize}

% =====================================================================
\section{Regras absolutas (nunca faz)}
\begin{itemize}
  \item Pergunta m\'edica (``que rem\'edio tomo?'') $\rightarrow$ ``o profissional responder\'a na consulta'' \sok
  \item Emerg\^encia $\rightarrow$ orienta SAMU (192) \sok
  \item Conv\^enio n\~ao credenciado $\rightarrow$ nega corretamente (s\'o oferece os credenciados) \sok
  \item N\~ao oferece desconto/condi\c{c}\~ao especial \sok
  \item N\~ao cancela consulta por conta pr\'opria (sem aprova\c{c}\~ao) \sparc
\end{itemize}

% =====================================================================
\section{Cadastro de paciente novo (conforme brief: S\'O o nome)}
\begin{itemize}
  \item Paciente novo (sem cadastro) que quer agendar $\rightarrow$ pede APENAS o nome \sok
  \item N\~ao pede CPF, data de nascimento nem cria conta/login (brief: s\'o ficha) \sok
  \item N\~ao pede cadastro se for s\'o d\'uvida (sem inten\c{c}\~ao de agendar) \sok
  \item Faz UMA pergunta por vez (n\~ao mistura nome + outras perguntas) \sok
  \item Grava o nome corretamente no cadastro (n\~ao fica vazio) \sok
  \item Paciente j\'a cadastrado N\~AO \'e perguntado de novo \sok
  \item Identifica\c{c}\~ao sempre por telefone normalizado (DDI 55) --- sem duplicar \sparc
\end{itemize}

% =====================================================================
\section{Agendamento com disponibilidade real}
\begin{itemize}
  \item Consulta a agenda real (n\~ao inventa hor\'ario) \sok
  \item Oferece no m\'aximo 2--3 datas/hor\'arios (n\~ao lista tudo) \sok
  \item Sempre mostra a DATA completa (dia-da-semana DD/MM), n\~ao s\'o ``quarta'' \sok
  \item ``pr\'oxima semana''/``pr\'oxima segunda'' $\rightarrow$ data correta da semana seguinte \sok
  \item Dia fora do hor\'ario (s\'abado, dia fechado) $\rightarrow$ informa e oferece datas v\'alidas \sok
  \item N\~ao oferece hor\'ario j\'a ocupado \sparc
  \item Ao confirmar, mant\'em o hor\'ario que o paciente j\'a informou (n\~ao repergunta) \sok
\end{itemize}

% =====================================================================
\section{Cria\c{c}\~ao do pedido (fim do fluxo)}
\begin{itemize}
  \item Ao confirmar (``sim/isso''), cria o pedido (n\~ao fica repetindo a confirma\c{c}\~ao) \sok
  \item Informa que ficou ``aguardando confirma\c{c}\~ao da cl\'inica'' \sok
  \item Pedido gravado no backend com a data/hora correta \sok
  \item Paciente provis\'orio criado com nome + telefone \sok
\end{itemize}

% =====================================================================
\section{Sincroniza\c{c}\~ao e fila de aprova\c{c}\~ao (cl\'inica)}
\begin{itemize}
  \item Pedido sincroniza para o Supabase (\texttt{ai\_appointment\_requests}) \sok
  \item Aparece na AGENDA (sidebar) na fila ``Pedidos de agendamento'' \sok
  \item Bolinha de notifica\c{c}\~ao no menu Agenda quando h\'a pendentes \sparc
  \item Cron sincroniza automaticamente (sem clicar ``Sincronizar'') --- depende da key no Vault \sfalta
  \item Bot\~ao ``Aprovar'' cria a consulta real na agenda \sok
  \item Bot\~ao ``Rejeitar'' marca como rejeitado \sok
\end{itemize}

% =====================================================================
\section{Avisos ao paciente (ap\'os decis\~ao da cl\'inica)}
\begin{itemize}
  \item Ao APROVAR $\rightarrow$ paciente recebe ``consulta CONFIRMADA para DD/MM HH:mm'' \sparc
  \item Ao REJEITAR $\rightarrow$ paciente recebe ``n\~ao conseguimos confirmar, quer outra data?'' \sparc
  \item (Depende do redeploy das Edge Functions no Supabase) \sfalta
\end{itemize}

% =====================================================================
\section{Remarca\c{c}\~ao}
\begin{itemize}
  \item Paciente pede para remarcar $\rightarrow$ oferece novos hor\'arios \sparc
  \item Mant\'em contexto (procedimento/consulta existente) durante a remarca\c{c}\~ao \sparc
  \item Atualiza a consulta com a nova data (n\~ao trava em loop) \sparc
\end{itemize}

% =====================================================================
\section{Automa\c{c}\~oes p\'os-agendamento}
\begin{itemize}
  \item Lembrete 24h antes (\texttt{max-lembretes}) --- testar com seu n\'umero \sparc
  \item Confirma\c{c}\~ao ap\'os agendar \sok
  \item Mensagem de retorno (X dias ap\'os \'ultima consulta) \sparc
  \item Anivers\'ario (com imagem anexa opcional) \sparc
  \item NPS p\'os-consulta \sparc
  \item Detec\c{c}\~ao de no-show 30min (\texttt{max-noshow}) --- falta deploy + cron \sfalta
  \item Resgate de inativos $>$60 dias (\texttt{max-resgate}) \sfalta
\end{itemize}

% =====================================================================
\section{Pr\'azo de aceite + escalada (pedido do Yuri)}
\begin{itemize}
  \item Pedido n\~ao aprovado em 2h $\rightarrow$ msg autom\'atica oferecendo remarcar \sfalta
  \item Ap\'os 3 tentativas sem aprova\c{c}\~ao $\rightarrow$ ``no momento indispon\'ivel, tente mais tarde'' \sfalta
\end{itemize}

% =====================================================================
\section{Pausar agendamentos (folga do m\'edico --- pedido do Yuri)}
\begin{itemize}
  \item M\'edico consegue pausar agendamentos de um dia (``amanh\~a estou de folga'') \sfalta
  \item IA n\~ao oferece hor\'ario nos dias pausados \sfalta
\end{itemize}

% =====================================================================
\section{Estabilidade / produ\c{c}\~ao}
\begin{itemize}
  \item Conex\~ao WhatsApp n\~ao cai sozinha (resolver 401/413) \sfalta
  \item Modelo segue bem as instru\c{c}\~oes (brief pede Claude/GPT-4o pago; hoje grátis) \sfalta
  \item Rota\c{c}\~ao de modelo funciona quando um atinge limite \sok
  \item Conversa de paciente real n\~ao some/some indevidamente (takeover) \sparc
\end{itemize}
